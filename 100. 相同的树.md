给你两棵二叉树的根节点 `p` 和 `q` ，编写一个函数来检验这两棵树是否相同。

如果两个树在结构上相同，并且节点具有相同的值，则认为它们是相同的。

```C++
class Solution {
public:
    bool check(TreeNode* _p,TreeNode* _q)
    {
        if(_p == nullptr && _q == nullptr)
        return true;
        if(_p == nullptr || _q == nullptr)
        return false;
        if(_p->val != _q->val)
        return false;
        return check(_p->left,_q->left) && check(_p->right,_q->right);
    }
    bool isSameTree(TreeNode* p, TreeNode* q) 
    {
        if(p == nullptr && q == nullptr)
        return true;
        return check(p,q);
    }
};
```
