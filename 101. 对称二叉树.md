给你一个二叉树的根节点 `root` ， 检查它是否轴对称。

思路：构造一个check函数来检查值这里要验证

```C++
class Solution {
public:
    bool check(TreeNode* _left,TreeNode* _right)
    {
        if(_left == nullptr && _right == nullptr)
        return true;
        if(_left == nullptr || _right == nullptr)
        return false;
        if(_right->val != _left->val)
        return false;
        return check(_left->left,_right->right) && check(_left->right,_right->left);
    }
    bool isSymmetric(TreeNode* root) 
    {
        if(root == nullptr)
        return true;
        return check(root->left,root->right);
    }
};
```
