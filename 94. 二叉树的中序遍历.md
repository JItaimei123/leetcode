给定一个二叉树的根节点 `root` ，返回 *它的 **中序** 遍历* 。

```C++
class Solution {
    private:
    void inorder(TreeNode* root,vector<int>& result)
    {
        if(root == nullptr)
        return;
        inorder(root->left,result);
        result.push_back(root->val);
        inorder(root->right,result);
    }
public:
    vector<int> inorderTraversal(TreeNode* root) 
    {
        vector<int> result;
        inorder(root,result);
        return result;
    }
};
```
