思路1：类似顺序表删除，从前往后遍历，遇到等于val的值，则把后⾯的元素往前挪动覆盖。这样

最坏情况下，⼤多数数据都等于val，时间复杂度就为 O(N2 )

```C
#include<stdio.h>
int removeElement(int* nums, int numsSize, int val)
{
	for (int i = 0; i < numsSize; i++)
	{
		if (val == nums[i])
		{
			for (int j = i; j < numsSize - 1; j++)
			{
				nums[j] = nums[j + 1];
			}
			numsSize--;
		}
	 }
		return numsSize;
}
int main() {
	int i = 0;
	int nums[] = { 3, 2, 2, 3 };
	int numsSize = sizeof(nums) / sizeof(nums[0]);
	int val = 3;
	int newsize = removeElement(nums, numsSize, 3);
	printf("%d ", newsize);
	return 0;
}
```

思路2：创建⼀个⼀样⼤⼩的数组tmp，遍历原数组把不等于val的元素依次挪动到tmp，再把数据

拷⻉回到原数组，这样时间复杂度就为 O(N) ，但是空间复杂度也是 O(N)

```C
#include<stdio.h>
#include<stdlib.h>
int removeElement(int* nums, int numsSize, int val)
{
	int* tmp = (int*)malloc(numsSize * sizeof(int));
	if(tmp ==NULL)
		return 1;
	int k = 0;
	for (int i = 0; i < numsSize; i++)
	{
		if (nums[i] != val)
		{
			tmp[k++] = nums[i];
		}
	}
	for (int i = 0; i < k; i++) 
	{
		nums[i] = tmp[i];
	}
	free(tmp);
	return k;//虽然val一直存在于nums中,但是返回k就可以控制无视其余多余的元素,假设如下图情况,如果在4,5的位置是val,那么返回k就是返回0-3的位置,4和5的位置上是val也没有影响.
}
int main() {
	int i = 0;
	int nums[] = { 3, 2, 2, 3 };
	int numsSize = sizeof(nums) / sizeof(nums[0]);
	int val = 3;
	int newsize = removeElement(nums, numsSize, val);
	printf("%d ", newsize);
	return 0;
}
```

思路3：我们可以在思路2基础上再改进，我们双指针(k和i)控制原地覆盖，src遍历元素数组，src找到不等于val的元素，覆盖到dst指向的原数组中的位置，这样时间复杂度就为 O(N) ，但是空间复杂度也是 O(1)

```C
#include<stdio.h>
int removeElement(int* nums, int numsSize, int val)
{
	int k = 0;
	for (int i = 0; i < numsSize; i++)
	{
		if (nums[i] != val)
		{
			nums[k++] = nums[i];
		}
	}
	return k;
}
int main() {
	int i = 0;
	int nums[] = { 3, 2, 2, 3 };
	int numsSize = sizeof(nums) / sizeof(nums[0]);
	int val = 3;
```
