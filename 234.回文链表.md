## 解题思路：

1. **快慢指针找中点**：`slow` 每次走一步，`fast` 每次走两步，当 `fast` 走到末尾时，`slow` 刚好在中点（或中点偏右）。

2. **反转后半部分**：从 `slow` 开始，把后半段链表反转。

3. **从头开始比较**：

  - `left` 从原链表头 `head` 出发

  - `right` 从反转后的后半段头 `prev` 出发

  - 依次比较 `left->val` 和 `right->val`

4. **返回值**：

  - 一旦有不相等 → 返回 `false`

  - 全部相等 → 返回 `true`

```C++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    bool isPalindrome(ListNode* head) 
    {
        if(!head || !head->next)
        return true;
        ListNode *fast = head,*slow = head;
        while(fast && fast->next)
        {
            slow = slow->next;
            fast = fast->next->next;
        }   
        ListNode * prev = nullptr;
        while(slow)
        {
            ListNode *next = slow->next;
            slow->next = prev;
            prev = slow;//更新表头
            slow = next;
        }
        ListNode * left = head,*right = prev;
        while(right && left)
        {
            if(left->val != right->val)
            return false;
            else
            {
                left = left->next;
                right = right->next;
            }
        }
        return true;
    }
};
```
