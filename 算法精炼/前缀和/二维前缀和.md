依旧是先预处理出来一个前缀和数组`dp`，规模与原来的数组一致

`dp[i][j]`代表从[1,1]到[i,j]位置，这段区间内的所有元素和

![image.png](./images/image.png)

假设要求的`dp[x1,y1] ~ dp[x2,y2]`在如图所示D的位置，那就要借助ABC来求解

具体步骤：

利用ABC辅助表达D，观察到`A+B`和`A+C`可以直接利用dp表达→`dp[x2][y2] = (A+B)+(A+C)+D-A`

→`D = dp[x2][y2]-(A+B)-(A+C)+A` = `dp[x2][y2]-dp[x1-1][y2]-dp[x2][y1-1]+dp[x1-1][y1-1]`

### 代码实现

```C++
#include <iostream>
#include <vector>
using namespace std;

int main() {
    // 1. 读入数据
    int n, m, q;
    cin >> n >> m >> q;
    vector<vector<int>> arr(n + 1, vector<int>(m + 1));
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            cin >> arr[i][j];
        }
    }

    // 2. 预处理前缀和矩阵
    vector<vector<long long>> dp(n + 1, vector<long long>(m + 1, 0));
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            dp[i][j] = dp[i - 1][j] + dp[i][j - 1] - dp[i - 1][j - 1] + arr[i][j];
        }
    }

    // 3. 处理查询
    while (q--) {
        int x1, y1, x2, y2;
        cin >> x1 >> y1 >> x2 >> y2;
        long long ans = dp[x2][y2] - dp[x1 - 1][y2] - dp[x2][y1 - 1] + dp[x1 - 1][y1 - 1];
        cout << ans << endl;
    }

    return 0;
}
```
