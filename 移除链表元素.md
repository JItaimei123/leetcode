给你一个链表的头节点 `head` 和一个整数 `val` ，请你删除链表中所有满足 `Node.val == val` 的节点，并返回 **新的头节点** 。

**示例 1：**

![removelinked-list.jpg](https://cdn2.flowus.cn/oss/9de70b1c-64af-48e9-b573-ac514e6565f0/removelinked-list.jpg?time=1784380500&token=2cbc40d2c01988fdd6ebcb3bd167388c618dfb1a3a326e24b4036a060854d71d&role=free)

```C++
输入：head = [1,2,6,3,4,5,6], val = 6
输出：[1,2,3,4,5]
```

**示例 2：**

```C++
输入：head = [], val = 1
输出：[]
```

**示例 3：**

```C++
输入：head = [7,7,7,7], val = 7
输出：[]
```

解题思路：

•准备⼀前⼀后两个指针prev和cur，遍历链表，当cur->val等于要删除的val时，prev指向cur的下⼀

个结点更改链接关系，释放cur结点

•在没有哨兵位的链表中，如果第⼀个结点的值等于val，则会导致prev为空，要单独判断处理更新

头指针；这⾥可以选择增加哨兵位结点，避免prev为空的情况。也可以不增加，两种解法。

```C++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* removeElements(ListNode* head, int val) 
    {
        struct ListNode * newNode = (struct ListNode *)malloc(sizeof(struct ListNode));
        newNode->next = head;
        struct ListNode * prev = newNode;
        struct ListNode * cur = head;
        while(cur)
        {
            if(cur->val == val)
            {
                prev->next = cur->next;
                cur = prev->next;
            }
            else
            {
                prev = cur;
                cur = cur->next;
            }
        }
        head = newNode->next;
        free(newNode);
        return head;
    }
};
```

```C++
public:
    ListNode* removeElements(ListNode* head, int val) 
    {
        struct ListNode*cur = head,*prev =NULL;
        while(cur)
        {
            if(cur->val ==val)

            if(prev == NULL)
            {
                head = cur->next;
                prev = cur;
                cur = head;
            }
            else
            {
               struct ListNode *next = cur->next;
               prev->next = cur->next;
               cur = next;
            }
            }
            else
            {
                prev = cur;
                cur = cur->next;
            }
        }
        return head;
    }
};
```
